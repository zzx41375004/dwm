# dwm快捷键

## 快捷键 ( mod = win )

~~~bash
mod + shift + d = 关闭dwm
mod + shift + x = 关闭lightdm
~~~

### 桌面相关操作

~~~bash
# 多显示器
mod + <             = 将焦点移动到左边的桌面
mod + >             = 将焦点移动到右边的桌面
mod + shift + <     = 将当前窗口移动到左边的桌面
mod + shift + >     = 将当前窗口移动到右边的桌面

# 一些开关
mod + enter         = 打开st
mod + g (google)    = 打开google-chrome
mod + y             = 打开dmenu
mod + k             = 小终端开关
mod + t             = 触摸板(如果有的话)开关
mod + o (opacity)   = 窗口渲染器开关 ( picom )

# 杂项
mod + f2/f1         = 调整亮度明/暗
mod + f6/f5         = 调整音量高/低
mod + f4            = 静音
mod + c             = 在~/Pictures/wallpapers里随机换壁纸
~~~

### 标签相关操作

~~~bash
mod + 0              = 临时展示所有窗口
mod + 1 ~ 9          = 标签 1 ~ 9
mod + q/w/f/p        = 标签 4/5/6/7
mod + shift + 标签号 = 将窗口移动至标签
# 例如当前窗口是google-chrome, 在第一个标签, 使用(mod+shift+q)即可将其移动至第四个标签
mod + b              = 状态栏开关
mod + space          = 切换模式, 平铺或全屏
~~~

### 窗口相关操作

~~~bash
mod + u/e或者n/i     = 在窗口栈中移动, 前进/后退(切换窗口)
mod + shift + u/e    = 将当前窗口向前/后移
mod + m              = 切换至主窗口
mod + l              = 关闭当前窗口
~~~
